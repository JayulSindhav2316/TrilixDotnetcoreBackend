SET @s = (SELECT IF(
    (SELECT COUNT(*)
        FROM INFORMATION_SCHEMA.COLUMNS
        WHERE table_name = 'documentobject'
        AND table_schema = DATABASE()
        AND column_name = 'EntityId'
    ) > 0,
    "SELECT 1",
    "ALTER TABLE `documentobject` ADD COLUMN `EntityId` INT NULL DEFAULT NULL"
));

PREPARE stmt FROM @s;
EXECUTE stmt;
DEALLOCATE PREPARE stmt;
