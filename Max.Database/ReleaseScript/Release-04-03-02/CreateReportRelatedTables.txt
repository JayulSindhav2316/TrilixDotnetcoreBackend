/* Create all report related tables */

CREATE TABLE `report` (
  `ReportId` INT NOT NULL AUTO_INCREMENT,
  `OrganizationId` INT NULL DEFAULT NULL,
  `Title` VARCHAR(45) NULL DEFAULT NULL,
  `Description` VARCHAR(250) NULL DEFAULT NULL,
  `UserId` INT NULL DEFAULT NULL,
  `ReportType` VARCHAR(45) NULL DEFAULT NULL,
  `isCommunity` INT NULL DEFAULT 0,
  `isFavorite` INT NULL DEFAULT 0,
  `LastUpdatedOn` DATETIME NULL DEFAULT NULL,
  `PreviewMode` INT NULL DEFAULT 0,
  `Fields` VARCHAR(500) NULL DEFAULT NULL,
  PRIMARY KEY (`ReportId`),
  INDEX `fk_report_organizationId_idx` (`OrganizationId` ASC) ,
  INDEX `fk_report_userId_idx` (`UserId` ASC) ,
  CONSTRAINT `fk_report_organizationId`
    FOREIGN KEY (`OrganizationId`)
    REFERENCES `organization` (`OrganizationId`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT,
  CONSTRAINT `fk_report_userId`
    FOREIGN KEY (`UserId`)
    REFERENCES `staffuser` (`UserId`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT);


CREATE TABLE `membershipreport` (
  `MembershipReportId` INT NOT NULL AUTO_INCREMENT,
  `Categories` VARCHAR(500) NULL DEFAULT NULL,
  `MembershipTypes` VARCHAR(500) NULL DEFAULT NULL,
  `Status` VARCHAR(45) NULL DEFAULT NULL,
  `ReportId` INT NULL DEFAULT NULL,
  PRIMARY KEY (`MembershipReportId`),
  INDEX `fk_membershipreport_report_id_idx` (`ReportId` ASC) ,
  CONSTRAINT `fk_membershipreport_report_id`
    FOREIGN KEY (`ReportId`)
    REFERENCES `report` (`ReportId`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT);


CREATE TABLE `reportfilter` (
  `ReportFilterId` INT NOT NULL AUTO_INCREMENT,
  `ReportId` INT NULL DEFAULT NULL,
  `Operator` VARCHAR(45) NULL DEFAULT NULL,
  `Value` VARCHAR(100) NULL DEFAULT NULL,
  `ReportFieldId` INT NULL DEFAULT NULL,
  PRIMARY KEY (`ReportFilterId`),
  INDEX `fk_reportfilter_report_id_idx` (`ReportId` ASC) ,
  CONSTRAINT `fk_reportfilter_report_id`
    FOREIGN KEY (`ReportId`)
    REFERENCES `report` (`ReportId`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT);

CREATE TABLE `reportshare` (
  `ReportShareId` INT NOT NULL AUTO_INCREMENT,
  `UserId` INT NULL DEFAULT NULL,
  `SharedToUserId` INT NULL DEFAULT NULL,
  `ReportId` INT NULL DEFAULT NULL,
  PRIMARY KEY (`ReportShareId`),
  INDEX `fk_report_share_report_id_idx` (`ReportId` ASC) ,
  CONSTRAINT `fk_report_share_report_id`
    FOREIGN KEY (`ReportId`)
    REFERENCES `report` (`ReportId`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT);


CREATE TABLE `reportsortorder` (
  `ReportSortOrderId` INT NOT NULL AUTO_INCREMENT,
  `FieldName` VARCHAR(100) NULL DEFAULT NULL,
  `Order` VARCHAR(45) NULL DEFAULT NULL,
  `FieldPathName` VARCHAR(250) NULL DEFAULT NULL,
  `ReportId` INT NULL DEFAULT NULL,
  PRIMARY KEY (`ReportSortOrderId`),
  INDEX `fk_reportsortorder_report_id_idx` (`ReportId` ASC) ,
  CONSTRAINT `fk_reportsortorder_report_id`
    FOREIGN KEY (`ReportId`)
    REFERENCES `report` (`ReportId`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT);


CREATE TABLE `reportcategory` (
  `ReportCategoryId` INT NOT NULL AUTO_INCREMENT,
  `Title` VARCHAR(100) NULL DEFAULT NULL,
  `Description` VARCHAR(250) NULL DEFAULT NULL,
  PRIMARY KEY (`ReportCategoryId`));


CREATE TABLE `reportfield` (
  `ReportFieldId` INT NOT NULL AUTO_INCREMENT,
  `ReportCategoryId` INT NULL DEFAULT NULL,
  `FieldName` VARCHAR(100) NULL DEFAULT NULL,
  `TableName` VARCHAR(100) NULL DEFAULT NULL,
  `Label` VARCHAR(100) NULL DEFAULT NULL,
  `FieldTitle` VARCHAR(100) NULL DEFAULT NULL,
  `DataType` VARCHAR(45) NULL DEFAULT NULL,
  `DisplayOrder` INT NULL DEFAULT NULL,
  `DisplayType` VARCHAR(45) NULL DEFAULT NULL,
  `CustomFieldId` INT NULL DEFAULT NULL,
  PRIMARY KEY (`ReportFieldId`),
  INDEX `fk_reportfield_customfield_id_idx` (`CustomFieldId` ASC) ,
  INDEX `fk_reportfield_reportcategory_id_idx` (`ReportCategoryId` ASC) ,
  CONSTRAINT `fk_reportfield_customfield_id`
    FOREIGN KEY (`CustomFieldId`)
    REFERENCES `customfield` (`CustomFieldId`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT,
  CONSTRAINT `fk_reportfield_reportcategory_id`
    FOREIGN KEY (`ReportCategoryId`)
    REFERENCES `reportcategory` (`ReportCategoryId`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT);
